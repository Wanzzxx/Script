-- Message counter
local messageCount = 0
local currentMode = "Console" -- "Console" or "RemoteSpy"
local blockedRemotes = {} -- Store blocked remote paths

-- Function to add a message to console
local function addMessage(text, messageType, remotePath)
	messageCount = messageCount + 1
	
	local messageFrame = Instance.new("Frame")
	messageFrame.Name = "Message_" .. messageCount
	messageFrame.Size = UDim2.new(1, -10, 0, 0)
	messageFrame.BackgroundTransparency = 1
	messageFrame.LayoutOrder = messageCount
	messageFrame.Parent = consoleFrame
	
	-- Block button (only show in RemoteSpy mode)
	local blockButton
	if currentMode == "RemoteSpy" and remotePath then
		blockButton = Instance.new("TextButton")
		blockButton.Size = UDim2.new(0, 30, 0, 20)
		blockButton.Position = UDim2.new(0, 0, 0, 0)
		blockButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
		blockButton.BackgroundTransparency = 0.9
		blockButton.Text = "[-]"
		blockButton.TextColor3 = Color3.fromRGB(255, 100, 100)
		blockButton.TextSize = 10
		blockButton.Font = Enum.Font.Code
		blockButton.Parent = messageFrame
		
		local blockCorner = Instance.new("UICorner")
		blockCorner.CornerRadius = UDim.new(0, 3)
		blockCorner.Parent = blockButton
		
		local blockStroke = Instance.new("UIStroke")
		blockStroke.Color = Color3.fromRGB(255, 100, 100)
		blockStroke.Thickness = 1
		blockStroke.Transparency = 0.7
		blockStroke.Parent = blockButton
	end
	
	-- Copy button
	local copyButton = Instance.new("TextButton")
	copyButton.Size = UDim2.new(0, 30, 0, 20)
	copyButton.Position = UDim2.new(0, blockButton and 35 or 0, 0, 0)
	copyButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	copyButton.BackgroundTransparency = 0.9
	copyButton.Text = "[/]"
	copyButton.TextColor3 = Color3.fromRGB(0, 255, 255)
	copyButton.TextSize = 10
	copyButton.Font = Enum.Font.Code
	copyButton.Parent = messageFrame
	
	local copyCorner = Instance.new("UICorner")
	copyCorner.CornerRadius = UDim.new(0, 3)
	copyCorner.Parent = copyButton
	
	local copyStroke = Instance.new("UIStroke")
	copyStroke.Color = Color3.fromRGB(0, 255, 255)
	copyStroke.Thickness = 1
	copyStroke.Transparency = 0.7
	copyStroke.Parent = copyButton
	
	-- Message label
	local messageLabel = Instance.new("TextLabel")
	messageLabel.Size = UDim2.new(1, blockButton and -70 or -35, 1, 0)
	messageLabel.Position = UDim2.new(0, blockButton and 70 or 35, 0, 0)
	messageLabel.BackgroundTransparency = 1
	messageLabel.Text = text
	messageLabel.TextSize = 14
	messageLabel.Font = Enum.Font.Code
	messageLabel.TextXAlignment = Enum.TextXAlignment.Left
	messageLabel.TextYAlignment = Enum.TextYAlignment.Top
	messageLabel.TextWrapped = true
	messageLabel.RichText = true
	messageLabel.Parent = messageFrame
	
	-- Color based on message type
	if messageType == "error" then
		messageLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
	elseif messageType == "warning" then
		messageLabel.TextColor3 = Color3.fromRGB(255, 220, 100)
	elseif messageType == "info" then
		messageLabel.TextColor3 = Color3.fromRGB(100, 200, 255)
	else
		messageLabel.TextColor3 = Color3.fromRGB(220, 220, 220)
	end
	
	-- Auto-resize frame based on text
	messageFrame.Size = UDim2.new(1, -10, 0, messageLabel.TextBounds.Y + 5)
	
	-- Block button functionality
	if blockButton and remotePath then
		blockButton.MouseButton1Click:Connect(function()
			if blockedRemotes[remotePath] then
				blockedRemotes[remotePath] = nil
				blockButton.Text = "[-]"
				blockButton.TextColor3 = Color3.fromRGB(255, 100, 100)
				blockStroke.Color = Color3.fromRGB(255, 100, 100)
			else
				blockedRemotes[remotePath] = true
				blockButton.Text = "[✓]"
				blockButton.TextColor3 = Color3.fromRGB(100, 255, 100)
				blockStroke.Color = Color3.fromRGB(100, 255, 100)
			end
		end)
		
		blockButton.MouseEnter:Connect(function()
			blockButton.BackgroundTransparency = 0.7
			blockStroke.Transparency = 0.4
		end)
		
		blockButton.MouseLeave:Connect(function()
			blockButton.BackgroundTransparency = 0.9
			blockStroke.Transparency = 0.7
		end)
	end
	
	-- Copy button functionality
	copyButton.MouseButton1Click:Connect(function()
		setclipboard(text)
		copyButton.Text = "✓"
		copyButton.TextColor3 = Color3.fromRGB(100, 255, 100)
		wait(1)
		copyButton.Text = "[/]"
		copyButton.TextColor3 = Color3.fromRGB(0, 255, 255)
	end)
	
	-- Copy button hover effects
	copyButton.MouseEnter:Connect(function()
		copyButton.BackgroundTransparency = 0.7
		copyStroke.Transparency = 0.4
	end)
	
	copyButton.MouseLeave:Connect(function()
		copyButton.BackgroundTransparency = 0.9
		copyStroke.Transparency = 0.7
	end)
	
	-- Scroll to bottom
	consoleFrame.CanvasPosition = Vector2.new(0, consoleFrame.AbsoluteCanvasSize.Y)
end

-- Function to clear console
local function clearConsole()
	for _, child in pairs(consoleFrame:GetChildren()) do
		if child:IsA("Frame") then
			child:Destroy()
		end
	end
	messageCount = 0
end

-- Hook into output
game:GetService("LogService").MessageOut:Connect(function(message, messageType)
	if currentMode == "Console" then
		if messageType == Enum.MessageType.MessageOutput then
			addMessage("[OUTPUT]: " .. message, "output")
		elseif messageType == Enum.MessageType.MessageInfo then
			addMessage("[INFO]: " .. message, "info")
		elseif messageType == Enum.MessageType.MessageWarning then
			addMessage("[WARNING]: " .. message, "warning")
		elseif messageType == Enum.MessageType.MessageError then
			addMessage("[ERROR]: " .. message, "error")
		end
	end
end)

-- RemoteSpy functionality
local function setupRemoteSpy()
	local mt = getrawmetatable(game)
	setreadonly(mt, false)
	
	local oldNamecall = mt.__namecall
	local oldIndex = mt.__index
	
	mt.__namecall = newcclosure(function(self, ...)
		local method = getnamecallmethod()
		local args = {...}
		
		if currentMode == "RemoteSpy" then
			if method == "FireServer" or method == "InvokeServer" then
				-- Get full remote path
				local remotePath = self:GetFullName()
				
				-- Check if this remote is blocked
				if blockedRemotes[remotePath] then
					return oldNamecall(self, ...)
				end
				
				-- Format arguments as a table string
				local argString = "local args = {\n"
				for i, v in pairs(args) do
					local valueStr = tostring(v)
					if type(v) == "string" then
						valueStr = '"' .. v .. '"'
					elseif type(v) == "table" then
						valueStr = "Table"
					end
					argString = argString .. "    [" .. i .. "] = " .. valueStr
					if i < #args then
						argString = argString .. ","
					end
					argString = argString .. "\n"
				end
				argString = argString .. "}\n\n"
				
				-- Add the remote path
				argString = argString .. remotePath
				
				addMessage(argString, "info", remotePath)
			end
		end
		
		return oldNamecall(self, ...)
	end)
	
	setreadonly(mt, true)
end

-- Setup RemoteSpy
pcall(setupRemoteSpy)
